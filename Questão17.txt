#include <stdio.h>
#include <stdlib.h>
void tururu(char **a, char **b, char **c, int atv){
    int *x, *y, *z, i;

    x=malloc(atv *sizeof (int));
    y=malloc(atv *sizeof (int));
    z=malloc(atv *sizeof (int));

    for(i=0; i<atv; i++){
        x[i]=0;
    }
    for(i=0; i<atv; i++){
        y[i]=0;
    }
    for(i=0; i<atv; i++){
        x[i]=rand()%20 -10;
    }
    printf("O valor do vetor ");
    printf("%s\n", a);
    for(i=0; i<atv; i++){
        printf("%d,", x[i]);
    }
    printf("\n");
    for(i=0; i<atv; i++){
        y[i]=rand()%20 -10;
    }
    printf("O valor do vetor ");
    printf("%s\n", b);
    for(i=0; i<atv; i++){
        printf("%d,", y[i]);
    }
    for(i=0; i<atv; i++){
        z[i]=0;
    }
    printf("\n");
    for(i=0; i<atv; i++){
        z[i]=x[i]+y[i];
    }
    printf("O valor do vetor ");
    printf("%s\n", c);
    for(i=0; i<atv; i++){
        printf("%d,", z[i]);
    }
    printf("\n");
    free(x);
    free(y);
    free(z);

}
int main()
{
    int tv;

    char **vetorum, **vetordois, **vetorsoma;
    vetorum = malloc( sizeof(char *)*10);
    vetordois = malloc( sizeof(char *)*10);
    vetorsoma= malloc( sizeof(char *)*10);

    printf("nome do primeiro vetor:" );
    gets(vetorum);

    printf("nome do segundo vetor: ");
    gets(vetordois);

    printf("nome do vetor soma:" );
    gets(vetorsoma);

    printf("tamanho do vetores \n");
    scanf("%d", &tv);

    (*tururu)(vetorum, vetordois, vetorsoma, tv);
    free(vetorum);
    free(vetordois);
    free(vetorsoma);

}
