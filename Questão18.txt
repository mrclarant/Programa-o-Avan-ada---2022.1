void multiplica(int **d, int **e, int  **f, int anla, int anca, int ancb){
    int i, j,x, p;
    int anlb, anlc, ancc;
         anlb=anla;
        anlc=anla;
        ancc=ancb;

    for(i = 0; i < anla; i++) {
        for(j = 0; j < anca; j++) {
            d[i][j]=rand()%10-5;


        }
    }


    for(i = 0; i < anca; i++) {
        for(j = 0; j < ancb; j++) {
            e[i][j]=rand()%10-5;

        }

    }




    for(i = 0; i < anla; i++) {
        for(j = 0; j < ancb; j++) {


            f[i][j]=0;
            for(p=0; p<anca; p++){
                x=x+d[i][p]*e[p][j];
            }
            f[i][j]=x;
            x=0;

        }

    }

}
