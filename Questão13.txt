#include <stdio.h>
#include <stdlib.h>


int main()
{
    int i,n,j;
    float temp=0.0f;
    printf("Valor de n= \n");
    scanf("%d", &n);


    float *a;
    a=malloc(n*sizeof(float));
    for(i=0;i<=n; i++){
        a[i]=rand() % 100+0.3;
    }


    printf("\n\n Antes \n\n");

    for(i= 0; i<=n; i++)
    {
        printf("%.2f  ", a[i]);
    }

    printf("\n");
    for(i=0;i<=n;i++)
    {
        for(j=0;j<n-i;j++)
        {
            if(a[j]>a[j+1])
            {
                temp=a[j+1];
                a[j+1]=a[j];
                a[j]=temp;
            }
        }
    }

    printf("\n\nDepois \n\n");

    for(i= 0; i<=n; i++)
    {
        printf("%.2f  ", a[i]);
    }
    printf("\n\n");
    free(a);
}
